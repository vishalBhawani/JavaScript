In JavaScript, an array is a special type of object used to store multiple values in a single variable.

// Creating an array
let fruits = ['Apple', 'Banana', 'Cherry'];
// Accessing elements
console.log(fruits[0]);
// Output: 'Apple'
// Modifying an element
fruits[1] = 'Orange';
console.log(fruits); //
Output: ['Apple', 'Orange', 'Cherry']
// Adding elements
fruits.push('Grapes');
console.log(fruits);
// Output: ['Apple', 'Orange', 'Cherry', 'Grapes']
// Length of an array
console.log(fruits.length);
// Output: 4
